72 /* Check if the inputted command corresponds to a legal command and return the comand id
73  * Note that command Id's do not start at "1",
74  * This is because values 1-3 are reserved for the following (in order):
75  * - "help"
76  * - "quit"
77  * - "history"
78  *
79  * @ command: The user inputted command
80  */
81 //[================================================================================================]
82 int check_command(std::string command)
83 {
84  // Need to set up better error messages
85  if(std::regex_match(command,find_tags)){return 4;}
86  else if(std::regex_match(command,find_files)){return 5;}
87  else if(std::regex_match(command,new_tags)){return 6;}
88  else if(std::regex_match(command,new_files)){return 7;}
89  else if(std::regex_match(command,new_file)){return 8;}
90  else if(std::regex_match(command,del_tags)){return 9;}
91  else if(std::regex_match(command,del_files)){return 10;}
92  else if(std::regex_match(command,del_file)){return 11;}
93  else if(std::regex_match(command,open_files)){return 12;}
94  else if(std::regex_match(command,close_files)){return 13;}
95  else if(std::regex_match(command,rename_tags)){return 14;}
96  else if(std::regex_match(command,rename_files)){return 15;}
97  else if(std::regex_match(command,get_attrs)){return 16;}
98  else if(std::regex_match(command,merge_1_1)){return 17;}
99  else if(std::regex_match(command,merge_m_1)){return 18;}
100  else if(std::regex_match(command,add_tags)){return 19;}
101  else if(std::regex_match(command,tag_files)){return 20;}
102  else if(std::regex_match(command,untag_file)){return 21;}
103  else if(std::regex_match(command,untag_files)){return 22;}
104  else if(std::regex_match(command,change_dir)){return 23;}
105  else{return 0;}
106 }
