402 void help()
403 {
404  print_cmnd_lst();
405  std::cout << "Please Enter '--' Followed By A Command Name To See Usage (Q/q to Quit): ";
406  std::string input;
407  std::cin >> input;
408  while(input != "Q" && input != "q")
409  {
410  if(input == "Q" || input == "q") break;
411  else
412  {
413  switch(check_usage(input))
414  {
415  case(UHELP):{print_help();break;}
416  case(UQUIT):{print_quit();break;}
417  case(UFIND):{print_find();break;}
418  case(UNEW):{print_new();break;}
419  case(UDEL):{print_del();break;}
420  case(UOPEN):{print_open();break;}
421  case(UCLOSE):{print_close();break;}
422  case(URNAME):{print_rname();break;}
423  case(UATTR):{print_attr();break;}
424  case(UMERG):{print_merge();break;}
425  case(UTAG):{print_tag();break;}
426  case(UUTAG):{print_utag();break;}
427  case(UCD):{print_cd();break;}
428  case(UREAD):{print_read();break;}
429  case(UWRITE):{print_write();break;}
430  case(UCOPY):{print_copy();break;}
431  default:
432  {
433  std::cout << "Please Enter '--' Followed By A Command Name (e.g. --help) To See Usage (Q/q to Quit): ";
434  }
435  }
436  }
437  input = "";
438  std::cout << "Please Enter '--' Followed By A Command Name To See Usage (Q/q to Quit): ";
439  std::cin >> input;
440  }
441  std::cout << std::endl;
442  return;
443 }
