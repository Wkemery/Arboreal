139 /* Check if the inputted command corresponds to a legal command and return the comand id
140  * Note that command Id's do not start at "1",
141  * This is because values 1-3 are reserved for the following (in order):
142  * - "help"
143  * - "quit"
144  * - "history"
145  *
146  * @ command: The user inputted command
147  */
148 //[================================================================================================]
149 int check_command(std::string command)
150 {
151  // Need to set up better error messages
152  if(std::regex_match(command,find_tags)){return FIND_TS;}
153  else if(std::regex_match(command,find_files)){return FIND_FS;}
154  else if(std::regex_match(command,new_tags)){return NEW_TS;}
155  else if(std::regex_match(command,new_files)){return NEW_FS;}
156  else if(std::regex_match(command,new_file)){return NEW_FP;}
157  else if(std::regex_match(command,del_tags)){return DEL_TS;}
158  else if(std::regex_match(command,del_files)){return DEL_FS;}
159  else if(std::regex_match(command,del_file)){return DEL_FP;}
160  else if(std::regex_match(command,open_files)){return OPEN_FP;}
161  else if(std::regex_match(command,open_file_cd)){return OPEN_F;}
162  else if(std::regex_match(command,close_files)){return CLOSE_FP;}
163  else if(std::regex_match(command,close_file_cd)){return CLOSE_F;}
164  else if(std::regex_match(command,rename_tags)){return RNAME_TS;}
165  else if(std::regex_match(command,rename_files)){return RNAME_FP;}
166  else if(std::regex_match(command,rename_file_cd)){return RNAME_FS;}
167  else if(std::regex_match(command,get_attrs)){return ATTR_FP;}
168  else if(std::regex_match(command,get_attr_cd)){return ATTR_FS;}
169  else if(std::regex_match(command,merge_1_1)){return MERG_1_1;}
170  else if(std::regex_match(command,merge_m_1)){return MERG_M_1;}
171  else if(std::regex_match(command,add_tags)){return TAG_FP;}
172  else if(std::regex_match(command,tag_files)){return TAG_FS;}
173  else if(std::regex_match(command,untag_file)){return UTAG_FP;}
174  else if(std::regex_match(command,untag_files)){return UTAG_FS;}
175  else if(std::regex_match(command,change_dir)){return CD_ABS;}
176  else if(std::regex_match(command,change_dir_rl)){return CD_RLP;}
177  else if(std::regex_match(command,read_x_path)){return READ_XP;}
178  else if(std::regex_match(command,read_x_cwd)){return READ_XCWD;}
179  else if(std::regex_match(command,read_path)){return READ_FP;}
180  else if(std::regex_match(command,read_cwd)){return READ_FCWD;}
181  else if(std::regex_match(command,write_path)){return WRITE_FP;}
182  else if(std::regex_match(command,write_cwd)){return WRITE_FCWD;}
183  else if(std::regex_match(command,write_x_path)){return WRITE_XFPF;}
184  else if(std::regex_match(command,write_x_cwd)){return WRITE_XFCWDF;}
185  else if(std::regex_match(command,copy_path)){return CPY_FP;}
186  else if(std::regex_match(command,copy_cwd)){return CPY_FCWD;}
187  else{return 0;}
188 }
